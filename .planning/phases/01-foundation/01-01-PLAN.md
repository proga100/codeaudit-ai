       ```typescript
       "use server";
       import { getDb, apiKeys } from "@codeaudit/db";
       import { encryptApiKey, decryptApiKey, maskApiKey } from "@codeaudit/db";
       import { validateApiKey } from "@/lib/api-key-validator";
       import { eq } from "drizzle-orm";

       export type Provider = "anthropic" | "openai" | "gemini";

       export async function addApiKey(provider: Provider, rawKey: string, label: string) {
         const validation = await validateApiKey(provider, rawKey);
         if (validation.status === "invalid") {
           return { success: false, error: validation.message };
         }
         const { encrypted, iv } = encryptApiKey(rawKey);
         const masked = maskApiKey(rawKey);
         const db = getDb();
         await db.insert(apiKeys).values({ provider, label, encryptedKey: encrypted, iv, maskedKey: masked });
         return { success: true };
       }

       export async function listApiKeys() {
         const db = getDb();
         return db.select({
           id: apiKeys.id,
           provider: apiKeys.provider,
           label: apiKeys.label,
           maskedKey: apiKeys.maskedKey,
           createdAt: apiKeys.createdAt,
         }).from(apiKeys).orderBy(apiKeys.provider, apiKeys.label).all();
       }

       export async function deleteApiKey(id: string) {
         const db = getDb();
         await db.delete(apiKeys).where(eq(apiKeys.id, id));
         return { success: true };
       }

       export async function updateApiKey(id: string, label: string) {
         const db = getDb();
         await db.update(apiKeys).set({ label, updatedAt: new Date() }).where(eq(apiKeys.id, id));
         return { success: true };
       }
       ```

       ```typescript
       #!/usr/bin/env node
       import { spawn } from "node:child_process";
       import path from "node:path";
       import { fileURLToPath } from "node:url";
       import fs from "node:fs";
       import os from "node:os";
       import crypto from "node:crypto";

       // Auto-generate ENCRYPTION_KEY on first run (Pitfall 2 from research)
       const envDir = path.join(os.homedir(), ".codeaudit");
       const envFile = path.join(envDir, ".env");
       if (!process.env["ENCRYPTION_KEY"]) {
         fs.mkdirSync(envDir, { recursive: true });
         let envContents = fs.existsSync(envFile) ? fs.readFileSync(envFile, "utf8") : "";
         if (!envContents.includes("ENCRYPTION_KEY=")) {
           const key = crypto.randomBytes(32).toString("hex");
           envContents += `\nENCRYPTION_KEY=${key}\n`;
           fs.writeFileSync(envFile, envContents, { mode: 0o600 });
         }
         // Load the key into this process
         const match = envContents.match(/ENCRYPTION_KEY=([^\n]+)/);
         if (match) process.env["ENCRYPTION_KEY"] = match[1].trim();
       }

       const PORT = process.env["PORT"] ?? "3000";
       const __dirname = path.dirname(fileURLToPath(import.meta.url));
       // In monorepo: CLI is at packages/cli/, Next.js app is at apps/web/
       const APP_DIR = path.resolve(__dirname, "../../apps/web");

       // Start Next.js dev server (use standalone server.js in production)
       const nextBin = path.join(APP_DIR, "node_modules", ".bin", "next");
       const server = spawn(nextBin, ["dev", "--port", PORT], {
         cwd: APP_DIR,
         env: { ...process.env, PORT },
         stdio: "inherit",
       });

       // Poll /api/health until server is ready, then open browser
       async function waitAndOpen() {
         const healthUrl = `http://localhost:${PORT}/api/health`;
         for (let i = 0; i < 30; i++) {
           try {
             const res = await fetch(healthUrl);
             if (res.ok) {
               const { default: open } = await import("open");
               await open(`http://localhost:${PORT}`);
               return;
             }
           } catch {
             // Not ready yet
           }
           await new Promise((r) => setTimeout(r, 1000));
         }
         console.error("Server did not become ready in 30s");
       }

       waitAndOpen().catch(console.error);

       process.on("SIGINT", () => { server.kill(); process.exit(0); });
       process.on("SIGTERM", () => { server.kill(); process.exit(0); });
       ```
