      createdAt: string;

    - completedAt: `audit.completedAt?.toISOString() ?? null`
    - createdAt: `audit.createdAt.toISOString()`

    Pass `groups` array to `<HistoryPage groups={groups} />`

    - Delete selected opens modal with `setShowDeleteConfirm("bulk")`

    - Each group: header row with folder icon (SVG), folder name in mono, "(N audits)" count, and Compare button

    - Row click: navigate to `/audit/${audit.id}/results`

    - Description: "This will permanently delete {N} selected audit(s) and all associated reports." for bulk, or "This will permanently delete this audit and all associated reports." for single
    - Warning: "This action cannot be undone." in `text-[12px] text-warning bg-warning-subtle p-2.5 rounded-lg`
